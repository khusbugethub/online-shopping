CREATE TABLE CAAMEDMAST
(
MEDIAID VARCHAR2(15) CONSTRAINT CAAMEDIAIDPK PRIMARY KEY,
MEDTYPE VARCHAR2(10)
);


CREATE TABLE CAAMEDDETAILS
(
MEDID VARCHAR2(15) REFERENCES CAAMEDMAST(MEDIAID),
MEDADVID VARCHAR2(12) CONSTRAINT CAAMEDADVIDPK PRIMARY KEY,
MEDADVNAME VARCHAR2(20),
MEDLANG VARCHAR2(10),
MEDCONTPERNAME VARCHAR2(25),
MEDCONTNUM VARCHAR2(25),
MEDADDR VARCHAR2(25),
MEDEMAIL VARCHAR2(20),
MEDFAX NUMBER(15),
MEDBAL NUMBER(12,2)
);


CREATE TABLE CAACUSTOMERS
(
CUSTID VARCHAR2(12) CONSTRAINT CAACUSTIDPK PRIMARY KEY,
CUSTNAME VARCHAR2(25),
CONTACTPERNAME VARCHAR2(25),
CONTNUM NUMBER(15),
CUSTADDR VARCHAR2(40),
PIN NUMBER(10),
EMAIL VARCHAR2(25),
FAX NUMBER(20),
CUSTTYPE VARCHAR2(10),
CUSTBAL NUMBER(12,2)
);


CREATE TABLE CAABOOKINGS
(
BKGID VARCHAR2(12) CONSTRAINT CAABKGIDPK  PRIMARY KEY,
BKGCUSTID VARCHAR2(12) REFERENCES CAACUSTOMERS(CUSTID),
BKGDOB DATE,
BKGDOP DATE,
BKGAMT NUMBER(12,2),
BKGSTATUS VARCHAR2(10),
CONTENT VARCHAR2(150)
);


CREATE TABLE CAANEWSPAPER
(
NPID VARCHAR2(12) REFERENCES CAAMEDDETAILS(MEDADVID), 
NPBKGID VARCHAR2(12) REFERENCES CAABOOKINGS(BKGID),
NPSIZE NUMBER(4),
NPADVPAGE VARCHAR2(25)
);


CREATE TABLE CAATV
(
TVCHANID VARCHAR2(12) REFERENCES CAAMEDDETAILS(MEDADVID),
TVBKGID VARCHAR2(12) REFERENCES CAABOOKINGS(BKGID),
TVDURATION NUMBER(3),
TVCNTTYPE VARCHAR2(10),
TVADVTIME VARCHAR2(10)
);



CREATE TABLE CAANPCHARGE
(
LANGNAME VARCHAR2(10),
RATEPSQCM NUMBER(7,2),
PAGE1PER NUMBER(2),
PAGE3 NUMBER(2),
PAGEMID NUMBER(2),
PAGELAST NUMBER(2)
);


CREATE TABLE CAATVCHARGE
(
LANGNAME VARCHAR2(10),
RATESCROLL NUMBER(6),
RATEANIM NUMBER(6),
RATEVIDEO NUMBER(6)
);


CREATE TABLE  CAAPAYMENTS
(
PMTID VARCHAR2(12) CONSTRAINT CAAPMTIDPK PRIMARY KEY,
PMTMEDID VARCHAR2(12),
PMTDOP DATE,
PMTMOP VARCHAR2(6),
PMTAMTPD NUMBER(12,2)
);


CREATE TABLE CAARECEIPTS
(
RCTID VARCHAR2(12) CONSTRAINT CAARCTIDPK PRIMARY KEY,
RCTCUSTID VARCHAR2(12),
RCTRD DATE,
RCTMOR VARCHAR2(6),
RCTAMTRCD NUMBER(12,2)
);


CREATE TABLE  CAAEMP
(
EMPID VARCHAR2(12) CONSTRAINT CAAEMPIDPK PRIMARY KEY,
EMPNAME VARCHAR2(20),
AGE NUMBER(2),
DOJ DATE,
ADDRESS VARCHAR2(40),
PNUMBER NUMBER(15)
);

CREATE  TABLE CAAID(ID VARCHAR2(15));


CREATE TABLE CAALOGIN
(
USERNAME VARCHAR2(20) CONSTRAINT CAAUSERNAMEPK PRIMARY KEY,
PASSWORD VARCHAR2(20), 
LOGINTYPE VARCHAR2(6)
);
